1- esPar, que devuelva si un número entero dado como parámetro es par o no.

        
        
let num = parseInt(prompt("Ingresa un nro: "));

function esPar() {

2- obtenerResto, que devuelva el resto del cociente entre dos números enteros dados como parámetros (sin usar el operador %).

        
        
let dividendo = parseInt(prompt("Ingresa el dividendo: "));
let divisor = parseInt(prompt("Ingresa el divisor: "));

function obtenerResto(dividendo, divisor) {
  let resto = dividendo;

  while (resto >= divisor) {
    resto -= divisor;
  }

  return resto;
}
console.log(obtenerResto(dividendo, divisor));

7- cantDivisores, que devuelva la cantidad de divisores que posee un número entero dado como parámetro.

        
        
let num = parseInt(prompt("Ingresa un número entero: "));

function cantDivisores(num) {
  if (num < 0) {
      num = num * -1;
  }
  let contDivisores = 0; //no 1 porque muestra 1 divisor +
  for (let i = 1; i <= num; i++) {
      if (num % i === 0) {
          contDivisores++;
      }
  }
  return contDivisores;
}
console.log(cantDivisores(num));

9- mostrarNPrimos, que muestre por la consola, separados por comas, los primeros n números primos. El valor de n se recibe como parámetro.

        
        
let n = parseInt(prompt("Ingresa un número entero: "));

function cantDivisores(numD) {
  if (numD < 0) {
    numD = numD * -1;
  }
  let divisores = [];
  for (let i = 1; i <= numD; i++) {
    if (numD % i === 0) {
      divisores.push(i);
    }
  }
  return divisores;
}

function esPrimo(num) {
  if (cantDivisores(num).length === 2) {
    return true;
  } else {
    return false;
  }
}

function mostrarNPrimos(n) {
  let cont = 1;
  let x = 2;
  let numerosPrimos = [];
  while (cont <= n) {
    if (esPrimo(x)) {
      numerosPrimos.push(x);
      cont++;
    }
    x++;
  }
  return numerosPrimos;
}

console.log(mostrarNPrimos(n));

10- esNumeroPerfecto, que devuelva si un número entero dado como parámetro es o no perfecto.

        
        
let perfecto = parseInt(prompt("Ingresa un número entero: "));

function esNumeroPerfecto(perfecto) {
  let divisores = cantDivisores(perfecto);
  let suma = 0;
  for (let i = 0; i < divisores.length; i++) {
    suma += divisores[i];
  }
  return suma === perfecto * 2;
}

console.log(esNumeroPerfecto(perfecto));

11- Realizá un programa que permita al usuario ingresar el valor unitario de cierto artículo y la cantidad de artículos vendidos por un vendedor, del cual se sabe que gana un sueldo fijo de $14000 más el 16% del monto total vendido. Con tales datos, la computadora debe calcular el monto a cobrar por el vendedor y mostrarlo.

        
        
let cantidadVentas = parseInt(prompt("Ingresa la cantidad de  productos vendidos: "));
let precioUnitario = parseFloat(prompt("Ingresa el precio unitario del producto vendido: "));

function salarioTotal(cantidadVentas, precioUnitario) {
  const salario = 14000;
  const comision = 0.16;
  return salarioTotal = salario + (cantidadVentas * precioUnitario * comision);
}

console.log(salarioTotal(cantidadVentas, precioUnitario));

13- Realizá un programa que permita al usuario ingresar la cantidad de cierto producto y el precio unitario de dicho producto. Por cada carga debe preguntar si se desea seguir ingresando de la forma "¿Deseás ingresar otro artículo? [S/N]". La carga de datos finaliza cuando el usuario lo determine. La computadora debe mostrar el monto total del ticket.

        
        
function preguntarSN(pregunta) {
  let respuesta;

  do {
    respuesta = prompt(`${pregunta} [S/N] `).toUpperCase();
  } while (respuesta !== "S" && respuesta !== "N");

  return respuesta === "S";
}

function calcularPromedio(listaNum) {
  let suma = 0;
  for (let i = 0; i < listaNum.length; i++) {
    suma += listaNum[i];
  }
  return suma / listaNum.length;
}

let listaProductos = [];
let ingresarMas = true;

while (ingresarMas) {
  let cantidadVentas = parseInt(prompt("Ingresa la cantidad de productos vendidos: "));
  let precioUnitario = parseFloat(prompt("Ingresa el precio unitario del producto vendido: "));

  listaProductos.push({cantidadVentas, precioUnitario});

  ingresarMas = preguntarSN("¿Deseás ingresar otro artículo?");
}

let total = 0;
let precios = [];

for (let i = 0; i < listaProductos.length; i++) {
  total += listaProductos[i].cantidadVentas * listaProductos[i].precioUnitario;
  precios.push(listaProductos[i].precioUnitario);
}

let precioPromedio = calcularPromedio(precios);

console.log(`El monto total del ticket es: $${total.toFixed(2)}`);
console.log(`El precio promedio de los productos es de $${precioPromedio.toFixed(2)}`);
