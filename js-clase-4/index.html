Definí las siguientes funciones, asegurándote su correcto funcionamiento con algunos tests. Asegurate de leer al menos dos veces los enunciados antes de intentar confeccionar las soluciones.

Definición y testeo de funciones

1- esPar, que devuelva si un número entero dado como parámetro es par o no.

        
        
  let num = parseInt(prompt("Ingresa un nro: "));

  function esPar(num) {
    if (num % 2 === 0 && num !== 0) {
      return `'${num}' es Par`;
    } else if (num === 0 || isNaN(num)){
        return "Ingresa un número válido";
    } else {
      return `'${num}' No es Par`;
    }
  }
  console.log(esPar(num))

2- obtenerResto, que devuelva el resto del cociente entre dos números enteros dados como parámetros (sin usar el operador %).

        
        
let dividendo = parseInt(prompt("Ingresa el dividendo: "));
let divisor = parseInt(prompt("Ingresa el divisor: "));

function obtenerResto(dividendo, divisor) {
  let resto = dividendo;

  while (resto >= divisor) {
    resto -= divisor;
  }

  return resto;
}
console.log(obtenerResto(dividendo, divisor));
