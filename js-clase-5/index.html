2- Realizá un programa que permita al usuario ingresar 10 números enteros. La computadora debe mostrar aquellos que sean múltiplos del último número ingresado.

        
        
function esMultiplo(num, divisor) {
  return num % divisor === 0;
}

function ultimoElemento(elementos) {
  return elementos[elementos.length - 1];
}

let numerosM = [];
for (let i = 0; i < 10; i++) {
  let numM = parseInt(prompt(`Ingrese el número ${i+1}:`));
  numerosM.push(numM);

  return Math.max(...numeros);

let limite = 9;
let consulta = "Ingresa caracter";

function cargarArray(limite) {
  let miArray = [];
  for (i = 0; i < limite; i++) {
    let elementos = prompt(`${consulta} ${i + 1}: `);
    miArray.push(elementos);
  }
  return miArray;
}

function invertirArray(array) {
  let arrayInvertido = [];
  for (let i = array.length - 1; i >= 0; i--) {
    arrayInvertido.push(array[i]);
  }
  return arrayInvertido;
}

let miArray = cargarArray(limite);
console.log(`Original: ${miArray}`);

let arrayInvertido = invertirArray(miArray);
console.log(`Invertido: ${arrayInvertido}`);

6- Realizá un programa que permita al usuario ingresar 9 caracteres. La computadora debe rotar un lugar hacia la derecha el contenido del array de caracteres sin usar un array auxiliar. Luego, mostrarlo.

        
        
let limite = 9;
let consulta = "Ingresa caracter";

function cargarArray(limite) {
  let miArray = [];
  for (i = 0; i < limite; i++) {
    let elementos = prompt(`${consulta} ${i + 1}: `);
    miArray.push(elementos);
  }
  return miArray;
}

function rotarArrayDerecha(rotado) {
  let ultimoElemento = rotado[rotado.length - 1];
  for (let i = rotado.length - 1; i >= 1; i--) {
    rotado[i] = rotado[i - 1];
  }
  rotado[0] = ultimoElemento;
  return rotado;
}

let miArray = cargarArray(limite);
console.log(`Original: ${miArray}`);

let arrayRotadoDerecha = rotarArrayDerecha(miArray);
console.log(`Rotado: ${arrayRotadoDerecha}`);
