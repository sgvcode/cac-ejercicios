Definición y testeo de funciones

1- Realizá un programa que permita al usuario ingresar 10 números enteros. La computadora debe mostrar el promedio de los mismos y cuáles fueron los valores ingresados que lo superaron.

        
        
let numeros = [];
let suma = 0;

for (let i = 0; i < 10; i++) {
  numeros.push(parseInt(prompt(`Ingrese el número ${i + 1}:`)));
}

// Suma, promedio y mayores al promedio
for (let i = 0; i < numeros.length; i++) {
  suma += numeros[i];
}

let promedio = suma / numeros.length;

let mayoresAlPromedio = [];

for (let i = 0; i < numeros.length; i++) {
  if (numeros[i] > promedio) {
    mayoresAlPromedio.push(numeros[i]);
  }
}

console.log(`El promedio de los números ingresados es: ${promedio}`);
console.log(`Los valores ingresados que superan el promedio son: ${mayoresAlPromedio.join(", ")}`);
